📅 Sheet 3: My Schedule

Purpose: Your active task list with due dates and status

Column Headers:

A1: "Task ID" B1: "Task Name" C1: "Category" D1: "Due Date" E1: "Status" F1: "Date Completed" G1: "Cost Actual" H1: "Vendor Used" I1: "Notes" J1: "Photo Link"

Formula for Auto-Populating:

// In A2 of My Schedule: =IF(ROW()-1<=COUNTA('Master Task List'!A:A)-1, INDEX('Master Task List'!A:A,ROW()),"") // In D2 (Due Date - auto-calculates based on frequency): =IF(B2="","", IF(E2="Done",F2, IF(INDEX('Master Task List'!E:E,MATCH(A2,'Master Task List'!A:A,0))="Monthly", EOMONTH(TODAY(),0), IF(INDEX('Master Task List'!E:E,MATCH(A2,'Master Task List'!A:A,0))="Quarterly", EOMONTH(TODAY(),3), IF(INDEX('Master Task List'!E:E,MATCH(A2,'Master Task List'!A:A,0))="Annual", DATE(YEAR(TODAY()),MONTH(INDEX('Master Task List'!J:J,MATCH(A2,'Master Task List'!A:A,0))),DAY(INDEX('Master Task List'!J:J,MATCH(A2,'Master Task List'!A:A,0)))), TODAY())))))

🤖 Optional: Apps Script Automation

Email Reminder Script:

function sendMaintenanceReminders() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("My Schedule"); var data = sheet.getDataRange().getValues(); var today = new Date(); var upcoming = []; // Find tasks due in next 7 days for (var i = 1; i < data.length; i++) { var dueDate = new Date(data[i][3]); var status = data[i][4]; var daysUntil = Math.floor((dueDate - today) / (1000 * 60 * 60 * 24)); if (daysUntil <= 7 && daysUntil >= 0 && status !== "Done") { upcoming.push({ task: data[i][1], category: data[i][2], due: data[i][3], daysLeft: daysUntil }); } } if (upcoming.length > 0) { var emailBody = "Upcoming Maintenance Tasks:\n\n"; for (var j = 0; j < upcoming.length; j++) { emailBody += upcoming[j].task + " (" + upcoming[j].category + ") - Due: " + Utilities.formatDate(upcoming[j].due, Session.getScriptTimeZone(), "MM/dd/yyyy") + " (" + upcoming[j].daysLeft + " days)\n"; } MailApp.sendEmail({ to: Session.getEffectiveUser().getEmail(), subject: "Maintenance Reminders: " + upcoming.length + " tasks due this week", body: emailBody }); } } // Set trigger to run weekly: function createTrigger() { ScriptApp.newTrigger('sendMaintenanceReminders') .timeBased() .everyWeeks(1) .onWeekDay(ScriptApp.WeekDay.MONDAY) .atHour(9) .create(); }
Note: Apps Script requires Google account and has daily email quotas (100 emails/day for free accounts). For production, include instructions but let users set up themselves.
